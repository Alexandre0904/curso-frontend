// Exercício 1



// Adição
let result = 10 + 15;
console.log(result); // 25
console.log(typeof result); // "number"

// Operações com strings e números
console.log("10" + 2); // "102" (string)
console.log("10" * 2); // 20 (number)
console.log("10" / 3); // 3.333... (float)
console.log("10" & 3); // 1 (number)
console.log(10 + true); // 11 (number)
console.log(10 == "10"); // true (boolean)
console.log(10 === "10"); // false (boolean)
console.log(10 < 11); // true (boolean)
console.log(10 > 12); // false (boolean)
console.log(10 <= 10.1); // true (boolean)
console.log(10 > 9.99); // true (boolean)
console.log(10 != "dez"); // true (boolean)
console.log("dez" + true); // "deztrue" (string)
console.log(10 * false); // 0 (number)
console.log(true + true); // 2 (number)

// Incremento e decremento
let num1 = 10;
num1++;
console.log(num1); // 11

let num2 = 10;
num2--;
console.log(num2); // 9

// Operadores bit a bit
console.log(1 & 1); // 1 (number)
console.log(1 & 0); // 0 (number)
console.log(0 & 0); // 0 (number)
console.log(1 | 0); // 1 (number)

// Outras operações
console.log(0 / 1); // 0 (number)
console.log(5 + 5 == 10); // true (boolean)
console.log("5" + "5" == 10); // false (boolean)
console.log("5" * 2 > 9); // true (boolean)
console.log((10 + 10) * 2); // 40 (number)
console.log(10 + 10 * 2); // 30 (number)

// Exercício 2



// a) branco == "branco"
var branco = "preto";
console.log(branco == "branco"); // false

// b) branco == cinza
var cinza = "branco";
console.log(branco == cinza); // false

// c) carro === branco
var carro = "preto";
console.log(carro === branco); // true (se ambos forem "preto")

// d) cavalo usando operador ternário
var cavalo = carro == "preto" ? "cinza" : "marron";
console.log(cavalo); // "cinza"

// e) Número de prestações para pagar o carro
var valor = 30000;
var entrada = 3000;
var prestacao = 750;
var valorFinanciado = valor - entrada;
var numeroPrestacoes = valorFinanciado / prestacao;
console.log(numeroPrestacoes); // 36

// f) Comprimento da string formada pelas variáveis de cores
var preto = "cinza";
var cores = branco + preto + cinza + carro;
console.log(cores.length); // 22 (sem contar espaços)
